## 利用channel阻塞实现顺序打印

利用无缓冲channel或容量为1的缓冲channel的阻塞特性实现顺序打印

```go
package main

import (
    "fmt"
    "sync"
    "time"
)

func main() {
    ch := make(chan int)
    var wg sync.WaitGroup

    for i := 1; i <= 10; i++ {
        wg.Add(1)
        go func(){
            defer wg.Done()
            n := <-ch
            fmt.Println(n)
            if n < 10 {
                ch <- n + 1
            }
        }()
    }

    ch <- 1 // 启动

    go func() {
        wg.Wait()
        close(ch)
    }()

    time.Sleep(time.Second)
}
```

## 通过互斥锁方式实现
```go
package main

import (
	"fmt"
	"sync"
)

func main() {
	var mu sync.Mutex
	var wg sync.WaitGroup
	num := 0
	for i := 0; i < 10; i++ {
		wg.Add(1)
		go func() {
			defer wg.Done()
			mu.Lock()
			defer mu.Unlock()
			num++
			fmt.Println(num)
		}()
	}

	wg.Wait()
}
```
