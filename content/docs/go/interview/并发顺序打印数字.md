## 利用channel阻塞实现顺序打印

利用无缓冲channel或容量为1的缓冲channel的阻塞特性实现顺序打印

```go
package main

import (
    "fmt"
    "sync"
    "time"
)

func main() {
    ch := make(chan int)
    var wg sync.WaitGroup

    for i := 1; i <= 10; i++ {
        wg.Add(1)
        go func(){
            defer wg.Done()
            n := <-ch
            fmt.Println(n)
            if n < 10 {
                ch <- n + 1
            }
        }()
    }

    ch <- 1 // 启动

    go func() {
        wg.Wait()
        close(ch)
    }()

    time.Sleep(time.Second)
}
```

## 通过互斥锁方式实现
```go
package main

import (
	"fmt"
	"sync"
)

func main() {
	var mu sync.Mutex
	var wg sync.WaitGroup
	num := 0
	for i := 0; i < 10; i++ {
		wg.Add(1)
		go func() {
			defer wg.Done()
			mu.Lock()
			defer mu.Unlock()
			num++
			fmt.Println(num)
		}()
	}

	wg.Wait()
}
```

## 顺序打印cat、dog、fish十遍

```go
package main

import (
	"fmt"
	"time"
)

func main() {
    catCh := make(chan int)
    dogCh := make(chan int)
    fishCh := make(chan int)
    // cat
    go func() {
        for range 10 {
            <-catCh
            fmt.Println("cat")
            dogCh <- 1
        }
    }()

    // dog
    go func() {
        for range 10 {
            <-dogCh
            fmt.Println("dog")
            fishCh <- 1
        }
    }()

    // fish
    go func() {
        for range 10 {
            <-fishCh
            fmt.Println("fish")
            catCh <- 1
        }
    }()

    // 启动
    catCh <- 1

    time.Sleep(time.Second)

}
```


```go
package main

import (
    "fmt"
    "time"
)

func main() {
    catCh := make(chan int)
    dogCh := make(chan int)
    fishCh := make(chan int)

    go print("cat", catCh, dogCh)
    go print("dog", dogCh, fishCh)
    go print("fish", fishCh, catCh)
    // 启动
    catCh <- 1

    time.Sleep(time.Second)

}

func print(s string, chReceive <-chan int, chSend chan<- int) {
    <-chReceive
    fmt.Println(s)
    chSend <- 1
}
```
