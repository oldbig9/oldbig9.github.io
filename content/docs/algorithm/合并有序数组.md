## 双指针倒序遍历

```go
package main

import "fmt"

func main() {
	s1 := []int{1, 3, 5, 7, 9}
	s2 := []int{2, 4, 6, 8, 10}
	fmt.Println(joinSlices(s1, s2))
}

// 合并有序数组
func joinSlices(nums1 []int, nums2 []int) []int {
	m, n, tail := len(nums1)-1, len(nums2)-1, len(nums1)+len(nums2)-1
	res := make([]int, len(nums1)+len(nums2))
	for ; m >= 0 || n >= 0; tail-- {
		var tmp int
		if m < 0 {
			tmp = nums2[n]
			n--
		} else if n < 0 {
			tmp = nums1[m]
			m--
		} else if nums1[m] > nums2[n] {
			tmp = nums1[m]
			m--
		} else {
			tmp = nums2[n]
			n--
		}
		res[tail] = tmp
	}
	return res
}
```
