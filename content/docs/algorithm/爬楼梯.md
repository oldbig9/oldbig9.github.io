## 递归+缓存中间结果

```go
package main

import "fmt"

func main() {
	m := make(map[int]int)
	for i := 1; i < 10; i++ {
		fmt.Println("num: ", climbStairs(i, m))
	}
}

// 爬楼梯
func climbStairs(n int, res map[int]int) int {
	if n <= 2 {
		return n
	}

	if v, ok := res[n]; ok {
		return v
	}

	return climbStairs(n-1, res) + climbStairs(n-2, res)
}
```
