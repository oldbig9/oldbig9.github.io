## 双指针

```go
// 最长子串
func lengthOfLongestSubstring(s string) int {
	max := 0
	m := make(map[byte]int)
	for left, right := 0, 0; right < len(s); right++ {
		if index, ok := m[s[right]]; ok && index >= left {
			left = index + 1
		}

		m[s[right]] = right
		if right-left+1 > max {
			max = right - left + 1
		}
	}
	return max
}

// 返回最长子串，这里可能存在多个长度一样的最长子串
func lengthOfLongestSubstringV2(s string) string {
	m := make(map[byte]int)
	res := ""
	for left, right := 0, 0; right < len(s); right++ {
		if index, ok := m[s[right]]; ok && index >= left {
			left = index + 1
		}

		m[s[right]] = right
		if right-left+1 > len(res) {
			res = s[left : right+1]
		}
	}
	return res
}
```
