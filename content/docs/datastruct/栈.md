栈是一种先入后出的数据结构，包含压栈、出栈操作

```go
package main

func main() {

}

type Stack struct {
    Data []int
}

func (s *Stack) Pop() int {
    if len(s.Data) == 0 {
        return -1
    }
    x := s.Data[len(s.Data)]
    s.Data = s.Data[len(s.Data)-1]

    return x
}

func (s *Stack) Push(x int) {
    s.Data = append(s.Data, x)
}

func (s *Stack) IsEmpty() bool {
    return len(s.Data) == 0
}

func (s *Stack) Peek() int {
    if len(s.Data) == 0 {
        return -1
    }

    return s.Data[0]
}

func (s *Stack) Size() int {
    return len(s.Data)
}

func (s *Stack) List() []int {
    return s.Data
}
```
